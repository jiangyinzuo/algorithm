### 求最长回文前缀（cf1326d）

可以利用kmp算法的get_next()方法
```c++
//
// Created by Jiang Yinzuo on 2020/4/8.
//

#include <iostream>
#include <string>
#include <algorithm>

int next[2000009] = {-1};

/**
 * 利用KMP算法中的get_next()求最大回文前缀
 * @param s 字符串
 * @return 最大回文前缀
 */
std::string find_palindrome(std::string &s) {
    // 构造一个原串 + ‘#’ + 原串的逆 的回文串
    std::string temp_s = s;
    std::reverse(temp_s.begin(), temp_s.end());
    temp_s += ('#' + s);

    // KMP中的get_next()方法
    int i = 0, j = -1;
    int len = temp_s.size();
    while (i < len) {
        if (j != -1 && temp_s[i] != temp_s[j]) {
            j = next[j];
        } else {
            next[++i] = ++j;
        }
    }

    return temp_s.substr(0, j);
}

void solve(std::string &s) {
    int len = s.length();
    int i = 0;
    int j = len - 1;
    for (; i < len; ++i, --j) {
        if (s[i] != s[j]) break;
    }
    std::string b = s.substr(0, i);
    std::cout << b;

    if (len > 2 * b.size()) {
        std::string s1 = s.substr(i, len - 2 * i);
        std::string res1 = find_palindrome(s1);
        std::reverse(s1.begin(), s1.end());
        std::string res2 = find_palindrome(s1);

        if (res1.size() > res2.size()) {
            std::cout << res1;
        } else {
            std::cout << res2;
        }
        std::reverse(b.begin(), b.end());
        std::cout << b;
    }
    std::cout << std::endl;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    int t;
    std::cin >> t;
    std::string s;
    while (t--) {
        std::cin >> s;
        solve(s);
    }
    return 0;
}
```
